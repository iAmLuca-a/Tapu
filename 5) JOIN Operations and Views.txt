CREATE TABLE department (
  dept_id NUMBER PRIMARY KEY,
  dept_name VARCHAR2(20)
);

CREATE TABLE employee (
  emp_id NUMBER PRIMARY KEY,
  name VARCHAR2(20),
  salary NUMBER,
  dept_id NUMBER,
  CONSTRAINT fk_dept FOREIGN KEY (dept_id)
  REFERENCES department(dept_id)
);
