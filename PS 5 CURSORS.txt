-- ============================================================
-- PROBLEM STATEMENT 5
-- Cursors: Implicit, Explicit, Cursor FOR Loop, Parameterized
-- ============================================================

-- ============================================================
-- CREATE TABLES
-- ============================================================

CREATE TABLE O_RollCall (
    roll_no NUMBER PRIMARY KEY,
    name    VARCHAR2(50)
);

CREATE TABLE N_RollCall (
    roll_no NUMBER PRIMARY KEY,
    name    VARCHAR2(50)
);

-- ============================================================
-- INSERT SAMPLE DATA
-- Roll 2 (Bob) exists in both => should be skipped during merge
-- Roll 3, 4 only in N_RollCall => should be inserted into O_RollCall
-- ============================================================

INSERT INTO O_RollCall VALUES (1, 'Alice');
INSERT INTO O_RollCall VALUES (2, 'Bob');

INSERT INTO N_RollCall VALUES (2, 'Bob');
INSERT INTO N_RollCall VALUES (3, 'Charlie');
INSERT INTO N_RollCall VALUES (4, 'Diana');

COMMIT;

SET SERVEROUTPUT ON;

-- ============================================================
-- 1. IMPLICIT CURSOR
-- Oracle creates this automatically for DML statements.
-- SQL%ROWCOUNT tells how many rows were affected.
-- ============================================================

BEGIN
    UPDATE O_RollCall SET name = UPPER(name);
    DBMS_OUTPUT.PUT_LINE('Implicit Cursor: Rows updated = ' || SQL%ROWCOUNT);
    COMMIT;
END;
/

-- ============================================================
-- 2. EXPLICIT CURSOR
-- You manually control: OPEN -> FETCH -> EXIT WHEN -> CLOSE
-- Use %NOTFOUND to detect when no more rows are available.
-- ============================================================

DECLARE
    CURSOR c_old IS SELECT roll_no, name FROM O_RollCall;
    v_roll O_RollCall.roll_no%TYPE;
    v_name O_RollCall.name%TYPE;
BEGIN
    DBMS_OUTPUT.PUT_LINE('--- Explicit Cursor Output ---');
    OPEN c_old;
    LOOP
        FETCH c_old INTO v_roll, v_name;
        EXIT WHEN c_old%NOTFOUND;
        DBMS_OUTPUT.PUT_LINE('Roll: ' || v_roll || ' | Name: ' || v_name);
    END LOOP;
    CLOSE c_old;
END;
/

-- ============================================================
-- 3. CURSOR FOR LOOP
-- Simplest cursor type. Oracle handles OPEN, FETCH, CLOSE.
-- Each iteration gives you a record variable automatically.
-- ============================================================

BEGIN
    DBMS_OUTPUT.PUT_LINE('--- Cursor FOR Loop Output ---');
    FOR rec IN (SELECT roll_no, name FROM N_RollCall) LOOP
        DBMS_OUTPUT.PUT_LINE('Roll: ' || rec.roll_no || ' | Name: ' || rec.name);
    END LOOP;
END;
/

-- ============================================================
-- 4. PARAMETERIZED CURSOR
-- Merge N_RollCall into O_RollCall.
-- Skip if roll_no already exists in O_RollCall.
-- Cursor accepts a parameter just like a function/procedure.
-- ============================================================

DECLARE
    -- Parameterized cursor: filters N_RollCall by roll_no
    CURSOR c_merge (p_roll NUMBER) IS
        SELECT roll_no, name
        FROM   N_RollCall
        WHERE  roll_no = p_roll;

    v_count  NUMBER;
    v_roll   N_RollCall.roll_no%TYPE;
    v_name   N_RollCall.name%TYPE;
BEGIN
    DBMS_OUTPUT.PUT_LINE('--- Parameterized Cursor: Merge Result ---');

    -- Loop through all records in N_RollCall
    FOR n_rec IN (SELECT roll_no, name FROM N_RollCall) LOOP

        -- Check if roll already exists in O_RollCall
        SELECT COUNT(*) INTO v_count
        FROM   O_RollCall
        WHERE  roll_no = n_rec.roll_no;

        IF v_count = 0 THEN
            -- Open parameterized cursor for this roll_no
            OPEN c_merge(n_rec.roll_no);
            FETCH c_merge INTO v_roll, v_name;

            INSERT INTO O_RollCall VALUES (v_roll, v_name);
            DBMS_OUTPUT.PUT_LINE('Inserted: ' || v_roll || ' - ' || v_name);

            CLOSE c_merge;
        ELSE
            DBMS_OUTPUT.PUT_LINE('Skipped (already exists): ' || n_rec.roll_no || ' - ' || n_rec.name);
        END IF;
    END LOOP;

    COMMIT;
    DBMS_OUTPUT.PUT_LINE('Merge complete.');
END;
/

-- Verify final state of O_RollCall
SELECT * FROM O_RollCall ORDER BY roll_no;
