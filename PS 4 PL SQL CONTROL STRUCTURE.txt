-- Enable output
SET SERVEROUTPUT ON;

-- Drop tables if already exist (optional)
DROP TABLE Fine;
DROP TABLE Borrower;

-- Create Borrower table
CREATE TABLE Borrower (
    Rollin      NUMBER(10)   PRIMARY KEY,
    Name        VARCHAR2(50) NOT NULL,
    DateofIssue DATE         NOT NULL,
    NameofBook  VARCHAR2(100) NOT NULL,
    Status      CHAR(1) DEFAULT 'I' CHECK (Status IN ('I','R'))
);

-- Create Fine table
CREATE TABLE Fine (
    Roll_no NUMBER(10),
    Date_   DATE,
    Amt     NUMBER(10,2),
    CONSTRAINT fk_roll FOREIGN KEY (Roll_no) REFERENCES Borrower(Rollin)
);

-- Sample data
INSERT INTO Borrower VALUES (101, 'Ravi',  TO_DATE('01-APR-2025','DD-MON-YYYY'), 'DBMS', 'I');
INSERT INTO Borrower VALUES (102, 'Priya', TO_DATE('10-MAR-2025','DD-MON-YYYY'), 'OS Concepts', 'I');
COMMIT;

-- ===================== PL/SQL BLOCK =====================

DECLARE
    v_roll     Borrower.Rollin%TYPE := &v_roll;
    v_book     Borrower.NameofBook%TYPE := '&v_book';
    v_doi      Borrower.DateofIssue%TYPE;
    v_days     NUMBER;
    v_fine_amt NUMBER := 0;
BEGIN
    -- Fetch borrower record
    SELECT DateofIssue INTO v_doi
    FROM Borrower
    WHERE Rollin = v_roll 
      AND NameofBook = v_book 
      AND Status = 'I';

    -- Calculate number of days since issue
    v_days := TRUNC(SYSDATE - v_doi);
    DBMS_OUTPUT.PUT_LINE('Days since issue: ' || v_days);

    -- Fine calculation
    IF v_days BETWEEN 15 AND 30 THEN
        v_fine_amt := v_days * 5;
        DBMS_OUTPUT.PUT_LINE('Fine (Rs 5/day): Rs ' || v_fine_amt);

    ELSIF v_days > 30 THEN
        v_fine_amt := (30 * 5) + ((v_days - 30) * 50);
        DBMS_OUTPUT.PUT_LINE('Fine (30 days @Rs5 + extra @Rs50): Rs ' || v_fine_amt);

    ELSE
        DBMS_OUTPUT.PUT_LINE('No fine applicable (within 15 days).');
    END IF;

    -- Update status
    UPDATE Borrower
    SET Status = 'R'
    WHERE Rollin = v_roll 
      AND NameofBook = v_book;

    IF SQL%ROWCOUNT = 0 THEN
        DBMS_OUTPUT.PUT_LINE('WARNING: No record updated.');
    ELSE
        DBMS_OUTPUT.PUT_LINE('Book return status updated to R.');
    END IF;

    -- Insert fine if applicable
    IF v_fine_amt > 0 THEN
        INSERT INTO Fine (Roll_no, Date_, Amt)
        VALUES (v_roll, SYSDATE, v_fine_amt);

        DBMS_OUTPUT.PUT_LINE('Fine record inserted.');
    END IF;

    COMMIT;

EXCEPTION
    WHEN NO_DATA_FOUND THEN
        DBMS_OUTPUT.PUT_LINE('ERROR: No issued book found for given Roll No and Book Name.');
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('ERROR: ' || SQLERRM);
        ROLLBACK;
END;
