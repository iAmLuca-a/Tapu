-- CREATE TABLES (with constraints)
-- ============================================================

CREATE TABLE Employee (
    employee_name VARCHAR2(50) PRIMARY KEY,
    street        VARCHAR2(100) NOT NULL,
    city          VARCHAR2(50)  NOT NULL
);

CREATE TABLE Company (
    company_name VARCHAR2(50) PRIMARY KEY,
    city         VARCHAR2(50) NOT NULL
);

CREATE TABLE Works (
    employee_name VARCHAR2(50),
    company_name  VARCHAR2(50),
    salary        NUMBER(10,2) NOT NULL,
    FOREIGN KEY (employee_name) REFERENCES Employee(employee_name),
    FOREIGN KEY (company_name)  REFERENCES Company(company_name)
);

CREATE TABLE Manages (
    employee_name VARCHAR2(50),
    manager_name  VARCHAR2(50),
    FOREIGN KEY (employee_name) REFERENCES Employee(employee_name)
);

-- QUERIES

-- Q1: Names of employees who work for First Bank Corporation
SELECT employee_name
FROM   Works
WHERE  company_name = 'First Bank Corporation';

-- Q2: Names and cities of FBC employees (JOIN Employee + Works)
SELECT e.employee_name, e.city
FROM   Employee e
JOIN   Works w ON e.employee_name = w.employee_name
WHERE  w.company_name = 'First Bank Corporation';

-- Q3: Name, street, city of FBC employees earning more than $10000
SELECT e.employee_name, e.street, e.city
FROM   Employee e
JOIN   Works w ON e.employee_name = w.employee_name
WHERE  w.company_name = 'First Bank Corporation'
  AND  w.salary > 10000;

-- Q4: Employees earning more than EVERY employee of Small Bank Corporation
-- ALL means the salary must exceed the HIGHEST salary in SBC
SELECT employee_name
FROM   Works
WHERE  salary > ALL (
    SELECT salary
    FROM   Works
    WHERE  company_name = 'Small Bank Corporation'
);

-- Q5: Employees earning more than their own company's average salary
-- Correlated subquery: inner query references w.company_name from outer query
SELECT w.employee_name
FROM   Works w
WHERE  w.salary > (
    SELECT AVG(w2.salary)
    FROM   Works w2
    WHERE  w2.company_name = w.company_name
);

-- Q6: Company with the smallest total payroll
-- NOTE: Nested aggregation MIN(SUM(...)) is Oracle-specific syntax
SELECT company_name
FROM   Works
GROUP BY company_name
HAVING SUM(salary) = (
    SELECT MIN(SUM(salary))
    FROM   Works
    GROUP BY company_name
);

-- Q7: Companies whose avg salary > avg salary at First Bank Corporation
SELECT company_name
FROM   Works
GROUP BY company_name
HAVING AVG(salary) > (
    SELECT AVG(salary)
    FROM   Works
    WHERE  company_name = 'First Bank Corporation'
);

-- Q8: View showing employee, company, and salary
CREATE VIEW emp_work_view AS
SELECT e.employee_name, w.company_name, w.salary
FROM   Employee e
JOIN   Works w ON e.employee_name = w.employee_name;

SELECT * FROM emp_work_view;

-- Q9: Create HighEarners table from employees earning above average
-- CREATE TABLE AS SELECT creates and fills the table in one statement
-- NOTE: This syntax is Oracle-specific; adjust for other databases if needed
CREATE TABLE HighEarners AS
SELECT employee_name, salary
FROM   Works
WHERE  salary > (SELECT AVG(salary) FROM Works);

SELECT * FROM HighEarners;

-- Q10: Delete employees who work for a company located in Gotham
-- FIX: The original query deletes from Employee directly, but Works and Manages
--      have FOREIGN KEY constraints referencing Employee. This causes a
--      FK constraint violation at runtime. Child table rows must be removed first.

-- Step 1: Identify Gotham employees (used across steps below)
-- Gotham employees: Jane Doe, Tom Hardy (they work for Gotham Corp, city = 'Gotham')

-- Step 2: Delete from Manages first (FK references Employee)
DELETE FROM Manages
WHERE employee_name IN (
    SELECT e.employee_name
    FROM   Employee e
    WHERE  e.city = 'Gotham'
);

-- Step 3: Delete from Works next (FK references Employee)
DELETE FROM Works
WHERE employee_name IN (
    SELECT e.employee_name
    FROM   Employee e
    WHERE  e.city = 'Gotham'
);

-- Step 4: Now safely delete from Employee
DELETE FROM Employee
WHERE city = 'Gotham';

COMMIT;
